// ============================================================
// CSV parsing
// ============================================================
function parseCSV(text) {
// strip BOM
if (text.charCodeAt(0) === 0xFEFF) text = text.slice(1);

// detect delimiter from header line (comma vs semicolon vs tab)
const firstLine = text.split(/\r?\n/, 1)[0] || '';
const counts = { ',': (firstLine.match(/,/g)||[]).length,
';': (firstLine.match(/;/g)||[]).length,
'\t': (firstLine.match(/\t/g)||[]).length };
const delim = Object.entries(counts).sort((a,b)=>b[1]-a[1])[0][0];

const rows = [];
let row = [], field = '', inQuotes = false;
for (let i = 0; i < text.length; i++) {
const c = text[i];
if (inQuotes) {
if (c === '"') {
if (text[i+1] === '"') { field += '"'; i++; }
else inQuotes = false;
} else field += c;
} else {
if (c === '"') inQuotes = true;
else if (c === delim) { row.push(field); field = ''; }
else if (c === '\n') { row.push(field); rows.push(row); row=[]; field=''; }
else if (c === '\r') { /* skip */ }
else field += c;
}
}
if (field.length || row.length) { row.push(field); rows.push(row); }

// remove trailing empty rows
while (rows.length && rows[rows.length-1].every(c => !c || !c.trim())) rows.pop();
if (!rows.length) return [];

const headers = rows[0].map(h => h.trim());
return rows.slice(1).map(r => {
const o = {};
headers.forEach((h, i) => { o[h] = (r[i] || '').trim(); });
return o;
});
}

// ============================================================
// Column normalization (Finnish + English aliases)
// ============================================================
const COLUMN_ALIASES = {
name: ['nimi', 'name', 'task', 'tehtava', 'tehtävä', 'title', 'otsikko'],
start: ['alkupaiva', 'alkupäivä', 'start', 'startdate', 'start_date', 'alku', 'alkaa'],
end: ['loppupaiva', 'loppupäivä', 'end', 'enddate', 'end_date', 'loppu', 'paattyy', 'päättyy'],
size: ['suuruusluokka', 'size', 'koko', 'tier'],
ticket: ['tikettinumero', 'ticket', 'ticketid', 'ticket_id', 'jira', 'tiketti'],
progress: ['progress', 'edistyminen', 'valmis', 'done', 'completion'],
};

function normalizeKey(k) {
return k.toLowerCase().replace(/[\s_\-]/g, '').replace(/ä/g,'a').replace(/ö/g,'o').replace(/å/g,'a');
}

function buildColumnMap(headers) {
const norm = headers.map(h => [h, normalizeKey(h)]);
const map = {};
for (const [field, aliases] of Object.entries(COLUMN_ALIASES)) {
const aliasSet = new Set(aliases.map(normalizeKey));
const found = norm.find(([_, n]) => aliasSet.has(n));
if (found) map[field] = found[0];
}
return map;
}

function normalizeRows(rawRows) {
if (!rawRows.length) return [];
const headers = Object.keys(rawRows[0]);
const map = buildColumnMap(headers);
if (!map.name || !map.start || !map.end) {
throw new Error('CSV:stä puuttuu pakollinen sarake. Tarvitaan: Nimi, Alkupaiva, Loppupaiva. Löydetyt sarakkeet: ' + headers.join(', '));
}
return rawRows
.filter(r => (r[map.name] || '').trim())
.map(r => ({
name: r[map.name].trim(),
start: parseDate(r[map.start]),
end: parseDate(r[map.end]),
size: ((r[map.size] || '') + '').trim().toUpperCase() || null,
ticket: ((r[map.ticket] || '') + '').trim() || null,
progress: map.progress !== undefined
? Math.min(100, Math.max(0, parseFloat(r[map.progress]) || 0))
: null,
}))
.filter(t => t.start && t.end);
}

// ============================================================
// Date helpers
// ============================================================
function parseDate(s) {
if (!s) return null;
s = ('' + s).trim();
// ISO: 2026-01-15
let m = /^(\d{4})-(\d{1,2})-(\d{1,2})$/.exec(s);
if (m) return new Date(+m[1], +m[2]-1, +m[3]);
// FI: 15.1.2026 or 15.01.2026
m = /^(\d{1,2})\.(\d{1,2})\.(\d{4})$/.exec(s);
if (m) return new Date(+m[3], +m[2]-1, +m[1]);
// US: 1/15/2026
m = /^(\d{1,2})\/(\d{1,2})\/(\d{4})$/.exec(s);
if (m) return new Date(+m[3], +m[1]-1, +m[2]);
// fallback
const d = new Date(s);
return isNaN(d) ? null : d;
}

function fmtDateFi(d) {
return `${d.getDate()}.${d.getMonth()+1}.${d.getFullYear()}`;
}
function diffDays(a, b) {
return Math.round((b - a) / 86400000);
}
function startOfMonth(d) { return new Date(d.getFullYear(), d.getMonth(), 1); }
function nextMonth(d) { return new Date(d.getFullYear(), d.getMonth()+1, 1); }
function quarterOf(d) { return Math.floor(d.getMonth() / 3) + 1; }

const MONTH_ABBR_FI = ['Tam','Hel','Maa','Huh','Tou','Kes','Hei','Elo','Syy','Lok','Mar','Jou'];
const MONTH_FULL_FI = ['Tammikuu','Helmikuu','Maaliskuu','Huhtikuu','Toukokuu','Kesäkuu',
'Heinäkuu','Elokuu','Syyskuu','Lokakuu','Marraskuu','Joulukuu'];
const DAY_ABBR_FI = ['Su','Ma','Ti','Ke','To','Pe','La']; // index = getDay()
const TICK_W = { day: 34, week: 72, month: 96 };

function startOfWeek(d) {
const delta = (d.getDay() === 0) ? 6 : d.getDay() - 1;
return new Date(d.getFullYear(), d.getMonth(), d.getDate() - delta);
}
function isoWeekNumber(d) {
const thu = new Date(d.getFullYear(), d.getMonth(), d.getDate() + 3 - ((d.getDay() + 6) % 7));
const yearStart = new Date(thu.getFullYear(), 0, 1);
return 1 + Math.round((thu - yearStart) / 604800000);
}
function selectScale(rawDays) {
if (rawDays <= 42) return 'day';
if (rawDays <= 180) return 'week';
return 'month';
}
function buildRangeBounds(minStart, maxEnd, scale) {
if (scale === 'month') {
return { rangeStart: startOfMonth(minStart), rangeEnd: nextMonth(maxEnd) };
}
return { rangeStart: startOfWeek(minStart), rangeEnd: addDays(startOfWeek(maxEnd), 7) };
}
function buildTicks(rangeStart, rangeEnd, scale) {
const ticks = [];
let cur = new Date(rangeStart);
const step = scale === 'day' ? 1 : scale === 'week' ? 7 : 0;
while (cur < rangeEnd) {
ticks.push(new Date(cur));
cur = step ? addDays(cur, step) : nextMonth(cur);
}
return ticks;
}

const SIZE_COLORS = {

function colorFor(size) {
return SIZE_HEX[size] || '#5b6470';
}

function darken(hex) {
return '#' + [1,3,5].map(i => Math.max(0, Math.round(parseInt(hex.slice(i,i+2),16) * 0.62))
.toString(16).padStart(2,'0')).join('');
}

function renderGantt(tasks) {
const svg = document.getElementById('chart');
svg.innerHTML = '';

// Compute timeline range, scale-aware
const minStart = tasks.reduce((a, t) => t.start < a ? t.start : a, tasks[0].start);
const maxEnd = tasks.reduce((a, t) => t.end > a ? t.end : a, tasks[0].end);
const rawDays = diffDays(minStart, maxEnd) + 1;
const scale = selectScale(rawDays);
const { rangeStart, rangeEnd } = buildRangeBounds(minStart, maxEnd, scale);
const totalDays = diffDays(rangeStart, rangeEnd);

// Layout constants
const leftCol = 280;
const headerRows = 56;
const rowH = 52;
const padRight = 24;

// Build tick list (one entry per column)
const tickW = TICK_W[scale];
const ticks = buildTicks(rangeStart, rangeEnd, scale);
const chartW = ticks.length * tickW;
const totalW = leftCol + chartW + padRight;
const totalH = headerRows + tasks.length * rowH + 24;

svg.setAttribute('width', totalW);
svg.setAttribute('height', totalH);
svg.setAttribute('viewBox', `0 0 ${totalW} ${totalH}`);

const ns = 'http://www.w3.org/2000/svg';

for (const [k, v] of Object.entries(attrs)) e.setAttribute(k, v);
if (text != null) e.textContent = text;
return e;
};

const defs = el('defs');
svg.appendChild(defs);

// Background for left column
svg.appendChild(el('rect', { x: 0, y: 0, width: leftCol, height: totalH, class: 'left-col-bg' }));
// Background for header band
svg.appendChild(el('rect', { x: leftCol, y: 0, width: chartW, height: headerRows, class: 'header-row-bg' }));

// Two-row header: major groups (top) + minor ticks (bottom)
const majorRowH = 24;
const minorRowH = headerRows - majorRowH; // 32

function getMajorKey(tick) {
if (scale === 'day') return `${tick.getFullYear()}-W${isoWeekNumber(tick)}`;
if (scale === 'week') return `${tick.getFullYear()}-${tick.getMonth()}`;
return `Q${quarterOf(tick)}-${tick.getFullYear()}`;
}
function getMajorLabel(tick) {
if (scale === 'day') return `V${isoWeekNumber(tick)} ${tick.getFullYear()}`;
if (scale === 'week') return `${MONTH_FULL_FI[tick.getMonth()]} ${tick.getFullYear()}`;
return `Q${quarterOf(tick)} ${tick.getFullYear()}`;
}
function getMinorLabel(tick) {
if (scale === 'day') return String(tick.getDate());
if (scale === 'week') return `V${isoWeekNumber(tick)}`;
return MONTH_ABBR_FI[tick.getMonth()];
}

// Top row: major group labels + dividers
let groupStart = 0;
for (let i = 0; i <= ticks.length; i++) {
const prev = ticks[i - 1], curr = ticks[i];
const sameGroup = prev && curr && getMajorKey(curr) === getMajorKey(prev);
if (i > 0 && (!sameGroup || i === ticks.length)) {
const x0 = leftCol + groupStart * tickW;
const w = (i - groupStart) * tickW;
svg.appendChild(el('text', {
x: x0 + w / 2, y: 16, class: 'axis-text quarter-label',
'text-anchor': 'middle'
}, getMajorLabel(prev)));
if (i < ticks.length) {
svg.appendChild(el('line', {
x1: leftCol + i * tickW, y1: 0,
x2: leftCol + i * tickW, y2: totalH,
class: 'grid-line quarter'
}));
}
groupStart = i;
}
}

// Bottom row: minor tick labels + grid lines
ticks.forEach((tick, i) => {
const x = leftCol + i * tickW;
if (i > 0) {
svg.appendChild(el('line', {
x1: x, y1: majorRowH, x2: x, y2: totalH,
class: 'grid-line month'
}));
}
svg.appendChild(el('text', {
x: x + tickW / 2, y: majorRowH + minorRowH / 2 + 4,
class: 'axis-text month-label',
'text-anchor': 'middle'
}, getMinorLabel(tick)));
});

// Header bottom border
svg.appendChild(el('line', {
x1: 0, y1: headerRows, x2: totalW, y2: headerRows,
class: 'left-col-divider'
}));
// Left column divider
svg.appendChild(el('line', {
x1: leftCol, y1: 0, x2: leftCol, y2: totalH,
class: 'left-col-divider'
}));

const dayToX = (d) => {
if (scale === 'month') {
// Month columns are equal-width but months have unequal day counts,
// so interpolate within each month rather than linearly across the whole range.
const idx = ticks.findIndex((t, i) =>
t <= d && (i === ticks.length - 1 || d < ticks[i + 1])
);
if (idx === -1) return leftCol;
const mStart = ticks[idx];
const mEnd = idx + 1 < ticks.length ? ticks[idx + 1] : rangeEnd;
return leftCol + (idx + diffDays(mStart, d) / diffDays(mStart, mEnd)) * tickW;
}
return leftCol + (diffDays(rangeStart, d) / totalDays) * chartW;
};

const today = new Date();
today.setHours(0, 0, 0, 0);

// Pass 1: row backgrounds (must come before weekend shading)
tasks.forEach((t, i) => {
const y = headerRows + i * rowH;
if (i % 2 === 1) {
svg.appendChild(el('rect', {
x: 0, y, width: totalW, height: rowH, class: 'row-bg-alt'
}));
}
svg.appendChild(el('line', {
x1: 0, y1: y + rowH, x2: totalW, y2: y + rowH, class: 'grid-line row'
}));
});

// Weekend column shading (day scale only)
if (scale === 'day') {
ticks.forEach((tick, i) => {
if (tick.getDay() === 0 || tick.getDay() === 6) {
svg.appendChild(el('rect', {
x: leftCol + i * tickW, y: headerRows,
width: tickW, height: totalH - headerRows,
fill: 'rgba(0,0,0,0.035)'
}));
}
});
}

// Today band (current period, all scales)
if (today >= rangeStart && today < rangeEnd) {
const todayIdx = ticks.findIndex((t, i) =>
t <= today && (i === ticks.length - 1 || today < ticks[i + 1])
);
if (todayIdx !== -1) {
svg.appendChild(el('rect', {
x: leftCol + todayIdx * tickW, y: 0,
width: tickW, height: totalH,
fill: 'rgba(192,57,43,0.06)'
}));
}
}

// Pass 2: row content
tasks.forEach((t, i) => {
const y = headerRows + i * rowH;

// Left column: task name + ticket / size badge
const nameY = y + rowH / 2 - 4;
const metaY = y + rowH / 2 + 12;
svg.appendChild(el('text', {
x: 20, y: nameY, class: 'row-name'
}, truncate(t.name, 32)));
const metaParts = [];
if (t.ticket) metaParts.push(t.ticket);
if (t.size) metaParts.push(`${t.size} · ${durationLabel(t.start, t.end)}`);
else metaParts.push(durationLabel(t.start, t.end));
svg.appendChild(el('text', {
x: 20, y: metaY, class: 'row-meta'
}, metaParts.join(' · ')));

// Bar
const x1 = dayToX(t.start);
const x2 = dayToX(addDays(t.end, 1));
const barX = Math.round(x1);
const barW = Math.max(6, Math.round(x2 - x1));
const barH = 26;
const barY = y + (rowH - barH) / 2;
const fill = colorFor(t.size);

// shadow
svg.appendChild(el('rect', {
x: barX, y: barY + 2, width: barW, height: barH,
rx: 5, ry: 5, class: 'bar-shadow'
}));
const bar = el('rect', {
x: barX, y: barY, width: barW, height: barH,
rx: 5, ry: 5, fill, class: 'bar', role: 'button'
});
svg.appendChild(bar);
bar.style.animationDelay = `${i * 0.04}s`;

// progress indicator: dark fill on done portion, white wash on remaining
if (t.progress !== null) {
const clipId = `bc${i}`;
const cp = el('clipPath', { id: clipId });
cp.appendChild(el('rect', { x: barX, y: barY, width: barW, height: barH, rx: 5, ry: 5 }));
defs.appendChild(cp);
const doneW = Math.round(barW * (t.progress / 100));
if (t.progress > 0) {
svg.appendChild(el('rect', {
x: barX, y: barY, width: Math.max(1, doneW), height: barH,
fill: darken(fill), 'clip-path': `url(#${clipId})`, 'pointer-events': 'none'
}));
}
if (t.progress < 100) {
svg.appendChild(el('rect', {
x: barX + doneW, y: barY, width: barW - doneW, height: barH,
fill: 'white', opacity: '0.48', 'clip-path': `url(#${clipId})`, 'pointer-events': 'none'
}));
}
}

// size badge on right end of bar
if (t.size && barW > 50) {
const badgeW = 22;
const badgeH = 16;
const bx = barX + barW - badgeW - 4;
const by = barY + (barH - badgeH) / 2;
svg.appendChild(el('rect', {
x: bx, y: by, width: badgeW, height: badgeH,
rx: 3, ry: 3, fill: 'rgba(0,0,0,0.18)', class: 'size-badge-bg'
}));
svg.appendChild(el('text', {
x: bx + badgeW / 2, y: by + badgeH / 2 + 4,
class: 'size-badge'
}, t.size));
}

// duration label inside or after bar
const days = diffDays(t.start, t.end) + 1;
const durTxt = `${days} pv`;
if (barW > 110) {
svg.appendChild(el('text', {
x: barX + 10, y: barY + barH/2 + 4, class: 'bar-label'
}, durTxt));
} else {
svg.appendChild(el('text', {
x: barX + barW + 8, y: barY + barH/2 + 4, class: 'bar-duration'
}, durTxt));
}

// tooltip handlers
bar.addEventListener('mouseenter', (e) => showTip(e, t));
bar.addEventListener('mousemove', (e) => moveTip(e));
bar.addEventListener('mouseleave', hideTip);
});

// Today line
if (today >= rangeStart && today <= rangeEnd) {
const tx = dayToX(today);
svg.appendChild(el('line', {
x1: tx, y1: 0, x2: tx, y2: totalH,
class: 'today-line'
}));
}

// Update subtitle and footer
document.getElementById('projectSubtitle').textContent =
`${tasks.length} työtä · ${fmtDateFi(minStart)} – ${fmtDateFi(maxEnd)}`;
document.getElementById('todayLabel').textContent = `Tänään: ${fmtDateFi(today)}`;
document.getElementById('footerNote').textContent =
`Luotu ${fmtDateFi(today)}`;

const hasProgress = tasks.some(t => t.progress !== null);
document.getElementById('progressLegendDivider').classList.toggle('hidden', !hasProgress);
document.getElementById('progressLegend').classList.toggle('hidden', !hasProgress);
}

function durationLabel(start, end) {
const days = diffDays(start, end) + 1;
if (days < 14) return `${days} pv`;
const weeks = Math.round(days / 7);
if (weeks < 9) return `${weeks} vk`;
const months = Math.round(days / 30.44);
return `${months} kk`;
}

function addDays(d, n) { return new Date(d.getFullYear(), d.getMonth(), d.getDate() + n); }

function truncate(s, n) {
if (!s) return '';
return s.length > n ? s.slice(0, n - 1) + '…' : s;
}

const tip = document.getElementById('tooltip');
function showTip(e, t) {
const days = diffDays(t.start, t.end) + 1;
tip.innerHTML = `
<strong>${escapeHtml(t.name)}</strong>
<div class="tt-row">${fmtDateFi(t.start)} – ${fmtDateFi(t.end)} · ${days} päivää</div>
${t.ticket ? `<div class="tt-row">Tiketti: ${escapeHtml(t.ticket)}</div>` : ''}
${t.size ? `<div class="tt-row">Suuruusluokka: ${escapeHtml(t.size)}</div>` : ''}
${t.progress !== null ? `<div class="tt-row">Edistyminen: ${t.progress} %</div>` : ''}
`;
tip.classList.add('show');
moveTip(e);
}
function moveTip(e) {
tip.style.left = e.clientX + 'px';
tip.style.top = e.clientY + 'px';
}
function hideTip() { tip.classList.remove('show'); }

function escapeHtml(s) {
return ('' + s).replace(/[&<>"']/g, c => ({
'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'
}[c]));
}

function loadFile(file) {
clearError();
if (!file) return;
const reader = new FileReader();
reader.onload = (e) => {
try {
const raw = parseCSV(e.target.result);
const tasks = normalizeRows(raw);
if (!tasks.length) {
showError('CSV:stä ei löytynyt yhtään validia tehtävää.');
return;
}
// Sort by start date
tasks.sort((a, b) => a.start - b.start);
renderGantt(tasks);
document.getElementById('chartContainer').classList.remove('hidden');
document.getElementById('legend').classList.remove('hidden');
document.getElementById('dropzone').classList.add('hidden');
document.getElementById('reloadBtn').disabled = false;
document.getElementById('printBtn').disabled = false;
document.getElementById('pngBtn').disabled = false;
} catch (err) {
console.error(err);
showError(err.message || 'CSV:n lukemisessa tapahtui virhe.');
}
};
reader.onerror = () => showError('Tiedoston lukeminen epäonnistui.');
reader.readAsText(file, 'UTF-8');
}

// Resolve CSS-variable colors to actual values for SVG export.
function inlineStyles(srcRoot, dstRoot) {
const srcAll = srcRoot.querySelectorAll('*');
const dstAll = dstRoot.querySelectorAll('*');
const PROPS = ['fill', 'stroke', 'font-family', 'font-size', 'font-weight'];
for (let i = 0; i < srcAll.length; i++) {
const cs = window.getComputedStyle(srcAll[i]);
const dst = dstAll[i];
PROPS.forEach(p => {
const v = cs.getPropertyValue(p);
if (v && v !== '' && v !== 'none' && v !== 'normal') {
dst.setAttribute(p, v.trim());
}
});
// remove class to avoid stylesheet dependency
dst.removeAttribute('class');
}
}
